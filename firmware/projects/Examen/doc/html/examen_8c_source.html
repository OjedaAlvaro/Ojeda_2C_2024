100void ControlarLEDs(void)
101{
102 // Tiene que fijarse que leds prender y apagar
103 if (distancia > 500)
104 {
105 LedOn(LED_1);
106 LedOff(LED_2);
107 LedOff(LED_3);
108 peligro = false;

183static void tareaAcelerometro(void *pvParameter)
184{
185 float suma=0;
186 while (true)
187 {
188 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
189 AnalogInputReadSingle(CH1, &ejeX);
190 AnalogInputReadSingle(CH2, &ejeY);
191 AnalogInputReadSingle(CH3, &ejeZ);
192
193 ejeX = (ejeX/0.3)-1.65;
194 ejeY = (ejeY/0.3)-1.65;
195 ejeZ = (ejeZ/0.3)-1.65;
196 suma= ejeX+ejeY+ejeZ;
197
198 if (suma> 4){
199 UartSendString(UART_CONNECTOR, "Caída detectada");
200 UartSendString(UART_CONNECTOR, "r\n");
