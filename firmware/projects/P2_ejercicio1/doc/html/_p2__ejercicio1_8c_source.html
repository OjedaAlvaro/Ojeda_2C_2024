72void ControlarLEDs(void){
73 // Si la distancia es menor a 10 cm, apaga todos los LEDs
74 if (distancia < 10) {
75 LedOff(LED_1);
76 LedOff(LED_2);
77 LedOff(LED_3);
78 }
79 // Si la distancia es mayor o igual a 10 cm y menor a 20 cm, prende solo el LED_1
80 else if (distancia < 20) {
81 LedOn(LED_1);
82 LedOff(LED_2);
83 LedOff(LED_3);
84 }
85 // Si la distancia es mayor o igual a 20 cm y menor a 30 cm, prende los LEDs 1 y 2

105static void MedirDistancia(void *pvParameter)
106{
107 while (true)
108 {
109 if (encendido)
110 {
111 // Medir la distancia en centimetros y actualizar la variable global
112 distancia = HcSr04ReadDistanceInCentimeters();
113 }
114 else
115 {
116 // Apagar el display LCD y todos los LEDs si la variable "encendido" es false
117 LcdItsE0803Off();
118 LedOff(LED_1);
119 LedOff(LED_2);
120 LedOff(LED_3);
121 }
122
123 // Esperar el tiempo especificado antes de medir de nuevo
124 vTaskDelay(DELAY_MEDIR / portTICK_PERIOD_MS);
125 }
126}
