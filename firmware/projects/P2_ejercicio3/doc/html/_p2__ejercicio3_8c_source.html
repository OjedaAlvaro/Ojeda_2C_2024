224void app_main(void)
225{
226 // Inicializo el LCD, LEDs,switches y el sensor
227 LcdItsE0803Init();
228 SwitchesInit();
229 LedsInit();
230 HcSr04Init(GPIO_3, GPIO_2);
231
232 timer_config_t timer_medir = {
233 .timer = TIMER_A,
234 .period = DELAY_MEDIR_uS,
235 .func_p = FuncTimerMedir,
236 .param_p = NULL};
237 TimerInit(&timer_medir);
238
239 timer_config_t timer_mostrar = {
240 .timer = TIMER_B,
241 .period = DELAY_LCD_uS,
242 .func_p = FuncTimerMostrar,
243 .param_p = NULL};
244 TimerInit(&timer_mostrar);
245 //inicializo el uart
246
247 serial_config_t uart_config = {
248 .port = UART_PC,
249 .baud_rate = 9600,
250 .func_p = FuncionTeclasUart,
251 .param_p = NULL
252 };
253
254 UartInit(&uart_config);
255
256
257 SwitchActivInt(SWITCH_1, InterrupcionTecla1, NULL);
258 SwitchActivInt(SWITCH_2, InterrupcionTecla2, NULL);
259
260 xTaskCreate(MostrarDisplayLCD, "MostrarDisplayLCD", 512, NULL, 5, &mostrar_task_handle);
261 xTaskCreate(MedirDistancia, "MedirDistancia", 512, NULL, 5, &medir_task_handle);
262
263 TimerStart(timer_medir.timer);
264 TimerStart(timer_mostrar.timer);
265}
